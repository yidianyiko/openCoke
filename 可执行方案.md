### 1.4 根因分析

```
根因树：
├── 消息卡住/丢失
│   ├── [根因1] handling 状态无超时检测
│   ├── [根因2] hold 状态无超时检测
│   └── [根因3] 锁超时时间与实际处理时间不匹配
│
├── 重复处理风险
│   ├── [根因4] 锁超时后原 Worker 不知道锁已失效
│   ├── [根因5] 释放锁时未验证锁归属
│   └── [根因6] 状态更新前未检查锁有效性
│
├── 消息打断问题
│   ├── [根因7] LLM 调用期间无法打断
│   ├── [根因8] 无 rollback 次数限制
│   └── [根因9] rollback 后历史记录不完整
│
├── 处理失败率高
│   ├── [根因10] Agno Model 层未配置重试
│   └── [根因11] 无错误分类机制
│
└── 锁泄漏风险
    ├── [根因12] 锁释放逻辑分散在多个分支
    └── [根因13] 异常路径可能遗漏锁释放
```

## 三、设计方案

### 2.1 新状态机（去掉 handling）

```
pending ──(获取锁)──→ [锁保护中] ──┬──(成功)──→ handled
                                   ├──(繁忙)──→ hold
                                   ├──(打断)──→ pending
                                   ├──(可重试错误)──→ pending (retry_count++)
                                   ├──(不可重试)──→ failed
                                   └──(进程崩溃)──→ 锁超时 → pending (自动恢复)

hold ──(空闲/超时)──→ pending
```

## 三、核心实现

### 3.1 安全锁释放（dao/lock.py）

```python
def release_lock_safe(self, resource_type, resource_id, lock_id) -> Tuple[bool, str]:
    """
    安全释放锁：只释放属于自己且未过期的锁
    
    解决：P4 - 锁超时后释放其他 Worker 的锁
    """
    resource_key = f"{resource_type}:{resource_id}"
    
    result = self.locks.delete_one({
        "resource_id": resource_key,
        "lock_id": lock_id,
        "expires_at": {"$gt": datetime.datetime.utcnow()}
    })
    
    if result.deleted_count > 0:
        return True, "released"
    
    # 检查失败原因
    existing_lock = self.locks.find_one({"resource_id": resource_key})
    if existing_lock is None:
        return False, "lock_not_found"
    if existing_lock.get("lock_id") != lock_id:
        return False, "lock_owned_by_other"
    return False, "lock_expired"
```


### 3.2 乐观锁更新（entity/message.py）

```python
def update_message_status_safe(message_id, new_status, expected_status="pending"):
    """
    乐观锁更新：只有当状态是预期值时才更新
    
    解决：P5 - 锁超时后多 Worker 同时更新状态

    result = _mongo.update_one(
```

### 3.4 Hold 检查（agent/runner/agent_background_handler.py 新增）

```python
HOLD_TIMEOUT = 3600  # 1小时

async def check_hold_messages():
    """检查 hold 状态消息，超时或角色空闲时恢复为 pending"""
    now = int(time.time())
    hold_messages = mongo.find_many("inputmessages", {"status": "hold"}, limit=100)
    
    for msg in hold_messages:
        try:
            relation = mongo.find_one("relations", {
                "uid": msg.get("from_user"),
                "cid": msg.get("to_user")
            })
            
            character_status = relation.get("character_info", {}).get("status", "空闲") if relation else "空闲"
            hold_started_at = msg.get("hold_started_at", now)
            is_timeout = (now - hold_started_at) > HOLD_TIMEOUT
            
            if character_status == "空闲" or is_timeout:
                mongo.update_one(
                    "inputmessages",
                    {"_id": msg["_id"]},
                    {"$set": {"status": "pending", "hold_started_at": None}}
                )
                logger.info(f"恢复 hold 消息: {msg['_id']}, reason={'timeout' if is_timeout else 'idle'}")
                
        except Exception as e:
            logger.error(f"检查 hold 消息失败: {e}")
```

## 四、Handler 核心修改

### 4.1 agent_handler.py 修改

```python
from entity.message import update_message_status_safe

MAX_RETRIES = 3
MAX_ROLLBACK = 3

def create_handler(worker_id: int = 0):
    worker_tag = f"[W{worker_id}]"
    
    async def _handler():
        input_messages = []
        lock = None
        lock_id = None
        conversation_id = None
        
        try:
            # ... 获取消息和锁的逻辑保持不变 ...
            
            # ========== 删除：不再标记为 handling ==========
            # for input_message in input_messages:
            #     input_message["status"] = "handling"  # 删除
            #     save_inputmessage(input_message)      # 删除
            
            # ========== 新增：检查重试次数 ==========
            valid_messages = []
            for msg in input_messages:
                if msg.get("retry_count", 0) >= MAX_RETRIES:
                    update_message_status_safe(msg["_id"], "failed", "pending")
                    logger.warning(f"{worker_tag} 消息达到最大重试次数: {msg['_id']}")
                    continue
                valid_messages.append(msg)
            
            input_messages = valid_messages
            if not input_messages:
                return
            
            # ... 处理逻辑 ...
            
            # ========== 新增：锁续期 ==========
            # Phase 2 前续期
            lock_manager.renew_lock("conversation", conversation_id, lock_id, timeout=120)
            
            async for event in streaming_chat_workflow.run_stream(...):
                if event["type"] == "message":
                    # 发送消息后续期
                    lock_manager.renew_lock("conversation", conversation_id, lock_id, timeout=120)
                    # ... 发送逻辑 ...
            
            # ========== 新增：rollback 次数限制 ==========
            if is_new_message_coming_in(...):
                max_rollback_count = max(msg.get("rollback_count", 0) for msg in input_messages)
                if max_rollback_count >= MAX_ROLLBACK:
                    logger.warning(f"{worker_tag} 达到最大 rollback 次数，强制处理")
                else:
                    is_rollback = True
                    for msg in input_messages:
                        mongo.update_one("inputmessages", {"_id": msg["_id"]}, 
                            {"$inc": {"rollback_count": 1}})
            
            # ========== 修改：使用乐观锁更新状态 ==========
            if is_finish:
                for msg in input_messages:
                    success = update_message_status_safe(msg["_id"], "handled", "pending")
                    if not success:
                        logger.warning(f"{worker_tag} 乐观锁更新失败，消息可能已被其他 Worker 处理: {msg['_id']}")
            
            if is_hold:
                for msg in input_messages:
                    mongo.update_one("inputmessages", {"_id": msg["_id"]},
                        {"$set": {"status": "hold", "hold_started_at": int(time.time())}})
                    
        except Exception as e:
            # ========== 简化：所有错误都重试，达到上限后标记 failed ==========
            logger.error(f"{worker_tag} 处理失败: {e}")
            
            for msg in input_messages:
                retry_count = msg.get("retry_count", 0) + 1
                
                if retry_count < MAX_RETRIES:
                    # 未达上限：保持 pending，增加 retry_count
                    mongo.update_one("inputmessages", {"_id": msg["_id"]},
                        {"$set": {"retry_count": retry_count, "last_error": str(e)[:500]}})
                else:
                    # 达到上限：标记 failed
                    update_message_status_safe(msg["_id"], "failed", "pending")
                    mongo.update_one("inputmessages", {"_id": msg["_id"]},
                        {"$set": {"last_error": str(e)[:500]}})
                    
        finally:
            # ========== 修改：使用安全锁释放 ==========
            if lock_id and conversation_id:
                released, reason = lock_manager.release_lock_safe(
                    "conversation", conversation_id, lock_id
                )
                if not released:
                    logger.warning(f"{worker_tag} 锁释放异常: {reason}")
    
    return _handler
```
